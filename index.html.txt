<body>

    <h2>Таблица с управлением строками</h2>

    <table id="targetTable">
        <thead>
            <tr>
                <th>Наименование</th>
                <th>Количество</th>
                <th>Действие</th>
            </tr>
        </thead>
        <tbody>
            <tr>
                <td><input type="text" style="width:90%"></td>
                <td><input type="number" style="width:90%"></td>
                <td><button class="btn-del" onclick="deleteRow(this)">Удалить</button></td>
            </tr>
        </tbody>
    </table>

    <button class="btn-add" onclick="addRow()">Добавить строку</button>

    <script>
        function addRow() {
            let tbody = document.getElementById("targetTable").getElementsByTagName('tbody')[0];
            let newRow = tbody.insertRow();

            let cell1 = newRow.insertCell(0);
            let cell2 = newRow.insertCell(1);
            let cell3 = newRow.insertCell(2);

            cell1.innerHTML = '<input type="text" style="width:90%">';
            cell2.innerHTML = '<input type="number" style="width:90%">';
            cell3.innerHTML = '<button class="btn-del" onclick="deleteRow(this)">Удалить</button>';
        }

        function deleteRow(button) {
            let row = button.parentNode.parentNode;
            row.parentNode.removeChild(row);
        }
    </script>
